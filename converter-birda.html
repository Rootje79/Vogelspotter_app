function processBirdaData() { const csvContent = document.getElementById('csvInput').value; if (!csvContent) return alert("Plak eerst de Birda data!"); try { const rawData = csvToJSON(csvContent); const convertedData = rawData.map(row => { const sciName = row.scientificName || ""; // ZOEKACTIE: We zoeken in jouw soorten.json op de Latijnse naam // omdat de Engelse commonName van Birda niet matcht met jouw NL lijst. const atlasInfo = vogelAtlas.find(v => v.WetSchap.toLowerCase() === sciName.toLowerCase() ); const nederlandseNaam = atlasInfo ? atlasInfo.naam : row.commonName; const status = atlasInfo ? atlasInfo.status : "Onbekend"; // Aantal bepalen: Als 'count' leeg is maar 'PRESENT', maken we er 1 van. let aantal = parseInt(row.count); if (isNaN(aantal)) aantal = 1; return { id: "BIRDA-" + row.sightingId, species: nederlandseNaam, // Nu in het Nederlands! latin: sciName, status: status, count: aantal, methode: "Gezien", tag: row.sessionTitle || "Birda Import", notes: (row.note || "").replace(/\n/g, " "), // Haal harde enters uit de Birda notes timestamp: new Date(row.date).toLocaleString('nl-NL'), isoDate: row.date, synced: true, coords: { lat: parseFloat(row.latitude), lon: parseFloat(row.longitude) } }; }); document.getElementById('jsonOutput').value = JSON.stringify(convertedData, null, 2); document.getElementById('downloadBtn').style.display = "block"; showStatus(`Succes! ${convertedData.length} Birda waarnemingen vertaald naar NL-formaat.`, "green"); } catch (err) { showStatus("Fout: " + err.message, "red"); } }
